set(MSCL_MICROSTRAIN_RTK_COMMANDS_FILES
    "${CMAKE_CURRENT_LIST_DIR}/ActivationCode.cpp"
    "${CMAKE_CURRENT_LIST_DIR}/ActivationCode.h"
    "${CMAKE_CURRENT_LIST_DIR}/DeviceStatusFlags.cpp"
    "${CMAKE_CURRENT_LIST_DIR}/DeviceStatusFlags.h"
)

# Filter and add the source files to the target
set(MSCL_MICROSTRAIN_RTK_COMMANDS_SOURCES ${MSCL_MICROSTRAIN_RTK_COMMANDS_FILES})
list(FILTER MSCL_MICROSTRAIN_RTK_COMMANDS_SOURCES INCLUDE REGEX "^.*\\.(c|cpp)$")
target_sources("${MSCL_LIBRARY}" PRIVATE "${MSCL_MICROSTRAIN_RTK_COMMANDS_SOURCES}")

# Filter and add the header files to the target using file sets
set(MSCL_MICROSTRAIN_RTK_COMMANDS_HEADERS ${MSCL_MICROSTRAIN_RTK_COMMANDS_FILES})
list(FILTER MSCL_MICROSTRAIN_RTK_COMMANDS_HEADERS INCLUDE REGEX "^.*\\.(h|hpp)$")
target_sources("${MSCL_LIBRARY}" PUBLIC
    FILE_SET "${MSCL_HEADER_FILE_SET}"
    FILES "${MSCL_MICROSTRAIN_RTK_COMMANDS_HEADERS}"
)
